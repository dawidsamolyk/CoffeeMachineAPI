package edu.issi.machine.subassembly;

import java.util.List;

import edu.issi.machine.Validator;
import edu.issi.machine.id.Identity;
import edu.issi.machine.id.ObjectWithIdentity;
import edu.issi.machine.operation.Operation;

/**
 * @author Dawid Samołyk
 * 
 *         Podzespół.
 */
public abstract class Subassembly extends ObjectWithIdentity {
    protected final List<Operation> operations;

    /**
     * @param id
     *            Identyfikator.
     * @param operations
     *            Operacje, które może wykonać wybrany podzespół.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli lista operacji będzie pusta lub niepoprawna.
     */
    public Subassembly(Identity id, List<Operation> operations) throws IllegalArgumentException {
	super(id);

	Validator.throwExceptionWhenContainsNullOrEmpty(operations,
		"Lista operacji nie może być pusta oraz nie może zawierać pustych operacji!");

	this.operations = operations;
    }

    /**
     * @param operation
     *            Operacja.
     * @return Czy podzespol wspiera podana operacje.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli operacja nie została utworzona.
     */
    public boolean supports(Operation operation) throws IllegalArgumentException {
	Validator.throwExceptionWhenObjectIsNotCreated(operation,
		"Nie jest możliwe sprawdzanie zgodności z pustą operacją!");

	return operations.contains(operation);
    }

    /**
     * Uruchomia podzespół.
     */
    public abstract void run();

    /**
     * Zatrzymuje podzespół.
     */
    public abstract void stop();

}
