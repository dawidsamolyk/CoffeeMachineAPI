package edu.issi.machine.configuration;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.NoSuchElementException;

import edu.issi.machine.Validator;
import edu.issi.machine.id.Identity;
import edu.issi.machine.operation.OperationStatus;
import edu.issi.machine.product.Product;
import edu.issi.machine.product.ingredient.Ingredient;

/**
 * @author Dawid
 *         Zamówienie.
 */
public class Order {
    protected Product product;

    /**
     * Konstruktor. Wymagana jest nazwa produktu, który ma być zamówiony. Ten
     * konstruktor powinien być używany do realizacji zamówienia produktu, który
     * samodzielnie definiuje użytkownik (tj. podaje składniki oraz ich
     * właściwości).
     * 
     * @param name
     *            Nazwa nowego produktu.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli nazwa produktu nie zostanie podana.
     */
    public Order(String name) throws IllegalArgumentException {
	Validator.throwExceptionWhenTextIsEmpty(name, "Nie podano nazwy produktu!");
	this.product = new Product(Identity.Factory.newIdentity(name));
    }

    /**
     * Konstruktor. Wymagany jest obiekt klasy Product, który będzie tworzony i
     * wydawany w ramach zamówienia. W tej klasie zapamiętana zostanie kopia
     * obiektu Product.
     * 
     * @param product
     *            Produkt.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli produkt nie został utworzony.
     */
    public Order(Product product) throws IllegalArgumentException {
	Validator.throwExceptionWhenObjectIsNotCreated(product, "Nie można złożyć zamówienia dla nieznanego produktu!");
	this.product = product.clone();
    }

    /**
     * Realizacja zamówienia.
     * 
     * @return Status wykonanych operacji.
     */
    public OperationStatus execute() {
	List<OperationStatus> result = new ArrayList<OperationStatus>();

	for (Ingredient eachIngredient : product) {
	    Collection<OperationStatus> operationsStatus = eachIngredient.doOperations();
	    result.addAll(operationsStatus);
	}

	return OperationStatus.Factory.getFrom(result);
    }

    /**
     * @author Dawid
     *
     */
    public class Configurator {

	/**
	 * @param index
	 *            Indeks, pod który ma zostać dodany składnik.
	 * @param ingredient
	 *            Składnik, który zostanie dodany do produktu.
	 * @throws UnsupportedOperationException
	 *             Wystąpi, jeśli podano nieprawidłowy indeks.
	 * @throws IllegalArgumentException
	 *             Wystąpi, jeśli obiekt wejściowy nie został utworzony.
	 * @see #Product.addAt(int, Ingredient)
	 */
	public void addAt(int index, Ingredient ingredient) throws UnsupportedOperationException, IllegalArgumentException {
	    product.addAt(index, ingredient);
	}

	/**
	 * @param index
	 *            Indeks, spod którego ma zostać usunięty składnik.
	 * @throws NoSuchElementException
	 *             Wystąpi, jeśli nie ma obiektu pod podanym indeksem.
	 * @see #Product.remove(int)
	 */
	public void remove(int index) throws NoSuchElementException {
	    product.remove(index);
	}

    }
}
