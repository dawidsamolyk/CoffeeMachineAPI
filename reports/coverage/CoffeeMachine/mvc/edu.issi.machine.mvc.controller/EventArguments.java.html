package edu.issi.machine.mvc.controller;

import edu.issi.machine.Validator;
import edu.issi.machine.mvc.view.View;

/**
 * @author DawidSamolyk
 *         Argumenty wydarzenia.
 */
public class EventArguments {
    private View caller;
    private String selectedElementName;

    /**
     * Konstruktor. Widok, który wywołał wydarzenie, jest wymagany.
     * 
     * @param caller
     *            Widok, który wywołał zdarzenie.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli nie zostanie podany widok, który wywołał
     *             zdarzenie.
     */
    public EventArguments(View caller) throws IllegalArgumentException {
	Validator.throwExceptionWhenObjectIsNotCreated(caller, "Widok wywołujący zdarzenie nie został ustawiony!");
	this.caller = caller;
    }

    /**
     * Konstruktor. Widok, który wywołał wydarzenie, jest wymagany. Również
     * nazwa wybranego elementu jest wymagana. Jest to dodatkowy argument, więc
     * jeśli nie jest konieczny, użyj innego konstruktora.
     * 
     * @param caller
     *            Widok, który wywołał zdarzenie.
     * @param selectedElementName
     *            Nazwa elementu, którego dotyczy wydarzenie.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli nie zostanie podany widok, który wywołał
     *             zdarzenie lub nie została podana nazwa elementu wybranego w
     *             zdarzeniu.
     */
    public EventArguments(View caller, String selectedElementName) throws IllegalArgumentException {
	this(caller);

	Validator.throwExceptionWhenTextIsEmpty(selectedElementName, "Nazwa wybranego w zdarzeniu elementu nie może być pusta!");
	this.selectedElementName = selectedElementName;
    }

    /**
     * Podaje informacje na temat tego czy podany widok wywołał zdarzenie, które
     * niesie ten argument.
     * 
     * @param view
     *            Widok.
     * @return Wskazuje czy widok jest tym, który wywołał to zdarzenie.
     * @throws IllegalArgumentException
     *             Wystąpi, jeśli porówywany widok nie został stworzony.
     */
    public boolean isCalledBy(View view) throws IllegalArgumentException {
	Validator.throwExceptionWhenObjectIsNotCreated(caller, "Widok wywołujący zdarzenie nie został ustawiony!");

	return view.equals(caller);
    }

    /**
     * Pobranie nazwy elementu.
     * 
     * @return Nazwa elementu, którego dotyczy zdarzenie.
     */
    public String getSelectedElementName() {
	return selectedElementName;
    }

    /**
     * Informuje czy podano dodatkowy argument.
     * 
     * @return Wskazuje czy posiada ustawioną nazwę elementu, którego dotyczy
     *         zdarzenie.
     */
    public boolean hasSelectedElementName() {
	return selectedElementName != null;
    }
}
